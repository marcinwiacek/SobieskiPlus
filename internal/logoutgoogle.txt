 link.setAttribute('google-signin-client_id', '<!--SIGN-IN-TOKEN-->');

 document.getElementsByTagName('head')[0].appendChild(link);

 function gapi_load() {
  gapi.load('auth2', function() {
   gapi.auth2.init({
    client_id: '<!--SIGN-IN-TOKEN-->'
   });
  });
 }
</script>
