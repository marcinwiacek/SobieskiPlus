   var pass = document.getElementById("confirmpass").value;

    alert('Hasła muszą być takie same');

   }
   http = createPOST();
   http.onload = function() {
    if (http.status != 200) {
     alert('verification error');
    } else {

    }
   };

    alert('verification error');
   };
   http.send("changepass=1&hash=" + encodeURIComponent(hash) + "&token=" + sha256(pass));

  };
