<script>
document.getElementById('confirmsub').disabled=true;
function sprawdzlogin() {
  if (document.getElementById("confirm").elements.namedItem("confirmuser").value=="" ||
     document.getElementById("confirm").elements.namedItem("confirmemail").value=="") {
    document.getElementById('confirmsub').disabled=true;
  } else {
    document.getElementById('confirmsub').disabled=false;
  }
}
document.getElementById("confirmuser").addEventListener("input", sprawdzlogin);
document.getElementById("confirmemail").addEventListener("input", sprawdzlogin);

document.confirm.onsubmit = function() {
  var user = document.getElementById("confirm").elements.namedItem("confirmuser").value; 
  var email = document.getElementById("confirm").elements.namedItem("confirmemail").value; 

  if (user == '' || email=='') return false;

  var http = new XMLHttpRequest();
  var url = window.location.protocol+'//'+window.location.hostname+':'+window.location.port+window.location.pathname;
  var params = "remind=1&token1="+sha256(hash+user)+"&token2="+sha256(hash+email);

  http.open('POST', url, true);
  http.setRequestHeader('Content-type', 'application/x-www-form-urlencoded');
  http.onload = function() {
    if (http.status != 200) {
      alert('verification error');
    } else {
      window.location.reload();
    }
  };
  http.onerror = function () {
    alert('verification error');
  };

  http.send(params);
  return false;
};
</script>
